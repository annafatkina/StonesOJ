cmake_minimum_required(VERSION 3.5)
project(prototype)

set(CMAKE_CXX_STANDARD 11)

set(SOURCE_FILES main_video.cpp)

add_executable(prototype ${SOURCE_FILES})

#setting up OpenCV

include_directories(
        "${PROJECT_SOURCE_DIR}/install/tf"
        "/usr/lib/python2.7/site-packages/tensorflow/include/external/nsync/public/"
        "/usr/lib/python2.7/site-packages/tensorflow/include/"
        "${PROJECT_SOURCE_DIR}/install/opencv/include"
        )

target_link_libraries(prototype
        "${PROJECT_SOURCE_DIR}/install/tf/libtensorflow_framework.so"
        "${PROJECT_SOURCE_DIR}/install/tf/libtensorflow_cc.so"
        "${PROJECT_SOURCE_DIR}/install/opencv/lib64/libopencv_world.so"
        )
